# 第三章 使用TypeScript改写部分接口

# 改写axios请求

import { ref } from 'vue'
import axios from 'axios'

// 添加一个参数作为要使用的 地址
function useURLLoader<T>(url: string) {
// 声明几个ref，代表不同的状态和结果
  const result = ref<T | null>(null)
  const loading = ref(true)
  const loaded = ref(false)
  const error = ref(null)

// 发送异步请求，获得data
// 由于 axios 都有定义，所以rawData 可以轻松知道其类型
  axios.get(url).then((rawData) => {
    loading.value = false
    loaded.value = true
    result.value = rawData.data
  }).catch((e) => {
    error.value = e
  })
  // 将这些ref 一一返回
  return {
    result,
    loading,
    error,
    loaded
  }
}


export {useURLLoader}

这样就可以在应用中使用，可以定义不同的数据类型，定义不同的接口。

# 改写新闻组件

<script lang="ts">
import { defineComponent } from "vue";
import {useURLLoader} from "@/services/urlLoader";

export interface NewsResult{
  content: string;
  date?: string;
  id: number;
  link: string;
  source?: string;
  time: string;
  title: string;
}

export default defineComponent({
  name: "News",
  setup() {
    const {result, loading, loaded} = useURLLoader<NewsResult[]>('http://127.0.0.1:8080/api/news')
    return {
      result,
    };
  },
});
</script>
